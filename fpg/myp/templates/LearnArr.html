// Deletion at index i
void deletionAtI(int *arr, int i){
    for(int j=i; j< n; j++){
        arr[j]=arr[j+1];
    }
}

//deletion after value num

void deletionAfter(int *arr, int num){
    int ans;
    for(int i=0; i< n; i++){
        if (arr[i]==num){
            ans=i;
        }
    }
    deletionAtI(arr, ans+1);
}

    void merge(int *arr, int s, int e){
        int mid=(s+e)/2;
        int m=mid+1-s;
        int n=e-mid;
        int M[m], N[n];
        int x=s;
        for(int i=0; i< m; i++){
            M[i]=arr[x++];
        }
    
        x=mid+1;
        for(int i=0; i< n; i++){
            N[i]=arr[x++];
        }
    
        int i=0;
        int j=0;
        int c=s;
        while(i< m && j< n){
            if(M[i]< N[j]){
                arr[c++]=M[i++];
            }
            else{
                arr[c++]=N[j++];
            }
        }
    
        while(i< m){
            arr[c++]=M[i++];
        }
        while(j< n){
            arr[c++]=N[j++];
        }
    }
    
    void mergesort(int *arr, int s, int e){
        int mid=(s+e)/2;
        if(s>=e){
            return;
        }
        mergesort(arr, s, mid);
        mergesort(arr, mid+1, e);

// You need to sort the array first

int binarySearch(int *arr,int s, int e int x){
    if(s>=e){
        return -1
    }
    int mid=(s+e)/2
    if(arr[mid]==x){
        return mid;
    }
    if(arr[mid]>x){
        binarySearch(arr, s, mid, x);
    }else{
        binarySearch(arr, mid+1, e, x);
    }
}

int kadane(int *arr, int n){
    int curr=0;
    int max= INT_MIN;

    for(int i=0; i< n; i++){
        curr+=arr[i];

        if(curr< 0){
            curr=0;
        }
        if(max< curr){
            max=curr;
        }
    }
    return max;
}
